55  template<typename JointModelDerived>
56  Model::JointIndex Model::addJoint(const Model::JointIndex parent,
57  const JointModelBase<JointModelDerived> & joint_model,
58  const SE3 & joint_placement,
59  const std::string & joint_name,
60  const Eigen::VectorXd & max_effort,
61  const Eigen::VectorXd & max_velocity,
62  const Eigen::VectorXd & min_config,
63  const Eigen::VectorXd & max_config
64  )
65  {
66  assert( (njoints==(int)joints.size())&&(njoints==(int)inertias.size())
67  &&(njoints==(int)parents.size())&&(njoints==(int)jointPlacements.size()) );
68  assert((joint_model.nq()>=0) && (joint_model.nv()>=0));
69 
70  assert(max_effort.size() == joint_model.nv()
71  && max_velocity.size() == joint_model.nv()
72  && min_config.size() == joint_model.nq()
73  && max_config.size() == joint_model.nq());
74 
75  Model::JointIndex idx = (Model::JointIndex) (njoints++);
76 
77  joints .push_back(JointModel(joint_model.derived()));
78  boost::get<JointModelDerived>(joints.back()).setIndexes(idx,nq,nv);
79 
80  inertias .push_back(Inertia::Zero());
81  parents .push_back(parent);
82  jointPlacements.push_back(joint_placement);
83  names .push_back(joint_name);
84  nq += joint_model.nq();
85  nv += joint_model.nv();
86 
87  // Optimal efficiency here would be using the static-dim bottomRows, while specifying the dimension in argument in the case where D::NV is Eigen::Dynamic.
88  // However, this option is not compiling in Travis (why?).
89  // As efficiency of Model::addJoint is not critical, the dynamic bottomRows is used here.
90  effortLimit.conservativeResize(nv);effortLimit.bottomRows(joint_model.nv()) = max_effort;
91  velocityLimit.conservativeResize(nv);velocityLimit.bottomRows(joint_model.nv()) = max_velocity;
92  lowerPositionLimit.conservativeResize(nq);lowerPositionLimit.bottomRows(joint_model.nq()) = min_config;
93  upperPositionLimit.conservativeResize(nq);upperPositionLimit.bottomRows(joint_model.nq()) = max_config;
94 
95  neutralConfiguration.conservativeResize(nq);
96  neutralConfiguration.tail(joint_model.nq()) = joint_model.neutralConfiguration();
97 
98  // Init and add joint index to its parent subtrees.
99  subtrees.push_back(IndexVector(1));
100  subtrees[idx][0] = idx;
101  addJointIndexToParentSubtrees(idx);
102  return idx;
103  }

123  int fidx)
124  {

297  inline void Data::computeLastChild (const Model & model)
298  {
299  typedef Model::Index Index;
300  std::fill(lastChild.begin(),lastChild.end(),-1);
301  for( int i=model.njoints-1;i>=0;--i )
302  {
303  if(lastChild[(Index)i] == -1) lastChild[(Index)i] = i;
304  const Index & parent = model.parents[(Index)i];
305  lastChild[parent] = std::max(lastChild[(Index)i],lastChild[parent]);
306 
307  nvSubtree[(Index)i]
308  = idx_v(model.joints[(Index)lastChild[(Index)i]]) + nv(model.joints[(Index)lastChild[(Index)i]])
309  - idx_v(model.joints[(Index)i]);
310  }
311  }

313  inline void Data::computeParents_fromRow (const Model & model)
314  {
315  for( Model::Index joint=1;joint<(Model::Index)(model.njoints);joint++)
316  {
317  const Model::Index & parent = model.parents[joint];
318  const int nvj = nv (model.joints[joint]);
319  const int idx_vj = idx_v(model.joints[joint]);
320 
321  if(parent>0) parents_fromRow[(Model::Index)idx_vj] = idx_v(model.joints[parent])+nv(model.joints[parent])-1;
322  else parents_fromRow[(Model::Index)idx_vj] = -1;
323  nvSubtree_fromRow[(Model::Index)idx_vj] = nvSubtree[joint];
324 
325  for(int row=1;row<nvj;++row)
326  {
327  parents_fromRow[(Model::Index)(idx_vj+row)] = idx_vj+row-1;
328  nvSubtree_fromRow[(Model::Index)(idx_vj+row)] = nvSubtree[joint]-row;
329  }
330  }
331  }
