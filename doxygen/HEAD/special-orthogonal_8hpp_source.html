58  template <class ConfigL_t, class ConfigR_t, class Tangent_t>
59  static void difference_impl(const Eigen::MatrixBase<ConfigL_t> & q0,
60  const Eigen::MatrixBase<ConfigR_t> & q1,
61  const Eigen::MatrixBase<Tangent_t> & d)
62  {
63  const_cast < Eigen::MatrixBase<Tangent_t>& > (d) [0]
64  = atan2 (q0(0)*q1(1) - q0(1)*q1(0), q0.dot(q1));
65  }

67  template <class ConfigIn_t, class Velocity_t, class ConfigOut_t>
68  static void integrate_impl(const Eigen::MatrixBase<ConfigIn_t> & q,
69  const Eigen::MatrixBase<Velocity_t> & v,
70  const Eigen::MatrixBase<ConfigOut_t> & qout)
71  {
72  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
73 
74  const Scalar & ca = q(0);
75  const Scalar & sa = q(1);
76  const Scalar & omega = v(0);
77 
78  Scalar cosOmega,sinOmega; SINCOS(omega, &sinOmega, &cosOmega);
79  // TODO check the cost of atan2 vs SINCOS
80 
81  out << cosOmega * ca - sinOmega * sa,
82  sinOmega * ca + cosOmega * sa;
83  const Scalar norm2 = q.squaredNorm();
84  out *= (3 - norm2) / 2;
85  }

87  template <class ConfigL_t, class ConfigR_t, class ConfigOut_t>
88  static void interpolate_impl(const Eigen::MatrixBase<ConfigL_t> & q0,
89  const Eigen::MatrixBase<ConfigR_t> & q1,
90  const Scalar& u,
91  const Eigen::MatrixBase<ConfigOut_t>& qout)
92  {
93  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
94 
95  assert ( (q0.norm() - 1) < 1e-8 && "initial configuration not normalized");
96  assert ( (q1.norm() - 1) < 1e-8 && "final configuration not normalized");
97  Scalar cosTheta = q0.dot(q1);
98  Scalar sinTheta = q0(0)*q1(1) - q0(1)*q1(0);
99  Scalar theta = atan2(sinTheta, cosTheta);
100  assert (fabs (sin (theta) - sinTheta) < 1e-8);
101 
102  if (fabs (theta) > 1e-6 && fabs (theta) < PI - 1e-6)
103  {
104  out = (sin ((1-u)*theta)/sinTheta) * q0
105  + (sin ( u *theta)/sinTheta) * q1;
106  }
107  else if (fabs (theta) < 1e-6) // theta = 0
108  {
109  out = (1-u) * q0 + u * q1;
110  }
111  else // theta = +-PI
112  {
113  double theta0 = atan2 (q0(1), q0(0));
114  out << cos (theta0 + u * theta),
115  sin (theta0 + u * theta);
116  }
117  }

150  template <class ConfigL_t, class ConfigR_t, class Tangent_t>
151  static void difference_impl(const Eigen::MatrixBase<ConfigL_t> & q0,
152  const Eigen::MatrixBase<ConfigR_t> & q1,
153  const Eigen::MatrixBase<Tangent_t> & d)
154  {
155  ConstQuaternionMap_t p0 (q0.derived().data());
156  ConstQuaternionMap_t p1 (q1.derived().data());
157  const_cast < Eigen::MatrixBase<Tangent_t>& > (d)
158  = log3((p0.matrix().transpose() * p1.matrix()).eval());
159  }

161  template <class ConfigIn_t, class Velocity_t, class ConfigOut_t>
162  static void integrate_impl(const Eigen::MatrixBase<ConfigIn_t> & q,
163  const Eigen::MatrixBase<Velocity_t> & v,
164  const Eigen::MatrixBase<ConfigOut_t> & qout)
165  {
166  ConstQuaternionMap_t quat(q.derived().data());
167  QuaternionMap_t quaternion_result (
168  (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived().data()
169  );
170 
171  Quaternion_t pOmega(exp3(v));
172  quaternion_result = quat * pOmega;
173  firstOrderNormalize(quaternion_result);
174  }
