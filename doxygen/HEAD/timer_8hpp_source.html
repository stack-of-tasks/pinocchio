31 struct StackTicToc
32 {
33  enum Unit { S = 1, MS = 1000, US = 1000000, NS = 1000000000 };
34  Unit DEFAULT_UNIT;
35  static std::string unitName(Unit u)
36  {
37  switch(u) { case S: return "s"; case MS: return "ms"; case US: return "us"; case NS: return "ns"; }
38  return "";
39  }

41  std::stack<struct timeval> stack;
42  mutable struct timeval t0;
43 
44 StackTicToc( Unit def = MS ) : DEFAULT_UNIT(def) {}
45 
46  inline void tic() {
47  stack.push(t0);
48  gettimeofday(&(stack.top()),NULL);
49  }
50 
51  inline double toc(const Unit factor)
52  {
53  gettimeofday(&t0,NULL);
54  double dt = (t0-stack.top())*factor;
55  stack.pop();
56  return dt;
57  }
58  inline void toc( std::ostream& os, double SMOOTH=1 )
59  {
60  os << toc(DEFAULT_UNIT)/SMOOTH << " " << unitName(DEFAULT_UNIT) << std::endl;
61  }
62 };
