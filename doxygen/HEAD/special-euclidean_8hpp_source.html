78  template <class ConfigIn_t, class Velocity_t, class ConfigOut_t>
79  static void integrate_impl(const Eigen::MatrixBase<ConfigIn_t> & q,
80  const Eigen::MatrixBase<Velocity_t> & vs,
81  const Eigen::MatrixBase<ConfigOut_t> & qout)
82  {
83  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
84  typedef Eigen::Matrix<Scalar, 2, 2> Matrix22;
85  typedef Eigen::Matrix<Scalar, 2, 1> Vector2;
86 
87  const double& c0 = q(2), s0 = q(3);
88  Matrix22 R0;
89  R0 << c0, -s0, s0, c0;
90 
91  const double& t = vs[2];
92  const double theta = std::fabs(t);
93 
94  if(theta > 1e-14)
95  {
96  // vs = [ x, y, t ]
97  // w = [ 0, 0, t ]
98  // v = [ x, y, 0 ]
99  // Considering only the 2x2 top left corner:
100  // Sp = [ 0, -1; 1, 0],
101  // if t > 0: S = Sp
102  // else : S = -Sp
103  // S / t = Sp / |t|
104  // S * S = - I2
105  // R = I2 + ( 1 - ct) / |t| * S + ( 1 - st / |t| ) * S * S
106  // = ( 1 - ct) / |t| * S + st / |t| * I2
107  //
108  // Ru = exp3 (w)
109  // tu = R * v = (1 - ct) / |t| * S * v + st / t * v
110  //
111  // M0 * Mu = ( R0 * Ru, R0 * tu + t0 )
112 
113  // FIXME remove this copy.
114  Vector2 v = vs.template head<2>();
115  Eigen::Matrix<Scalar,2,1> cst;
116  SINCOS (t, &cst[1], &cst[0]);
117  const Scalar inv_theta = 1/theta;
118  const Scalar c_coeff = (1.-cst[0]) * inv_theta;
119  const Scalar s_coeff = std::fabs(cst[1]) * inv_theta;
120  const Vector2 Sp_v (-v[1], v[0]);
121 
122  if (t > 0) out.template head<2>() = q.template head<2>() + R0 * (s_coeff * v + c_coeff * Sp_v);
123  else out.template head<2>() = q.template head<2>() + R0 * (s_coeff * v - c_coeff * Sp_v);
124  out.template tail<2>() = R0 * cst;
125  }
126  else
127  {
128  // cos(t) ~ 1 - t^2 / 2
129  // sin(t) ~ t
130  out.template head<2>() = q.template head<2>() + R0*vs.template head<2>();
131  out(2) = c0 * 1 - s0 * t;
132  out(3) = s0 * 1 + c0 * t;
133  }
134  }

195  template <class ConfigL_t, class ConfigR_t, class Tangent_t>
196  static void difference_impl(const Eigen::MatrixBase<ConfigL_t> & q0,
197  const Eigen::MatrixBase<ConfigR_t> & q1,
198  const Eigen::MatrixBase<Tangent_t> & d)
199  {
200  ConstQuaternionMap_t p0 (q0.derived().template tail<4>().data());
201  ConstQuaternionMap_t p1 (q1.derived().template tail<4>().data());
202  const_cast < Eigen::MatrixBase<Tangent_t>& > (d)
203  = log6( SE3(p0.matrix(), q0.derived().template head<3>()).inverse()
204  * SE3(p1.matrix(), q1.derived().template head<3>())).toVector();
205  }

207  template <class ConfigIn_t, class Velocity_t, class ConfigOut_t>
208  static void integrate_impl(const Eigen::MatrixBase<ConfigIn_t> & q,
209  const Eigen::MatrixBase<Velocity_t> & v,
210  const Eigen::MatrixBase<ConfigOut_t> & qout)
211  {
212  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
213  ConstQuaternionMap_t quat(q.derived().template tail<4>().data());
214  QuaternionMap_t res_quat (out.template tail<4>().data());
215 
216  SE3 M0 (quat.matrix(), q.derived().template head<3>());
217  SE3 M1 (M0 * exp6(Motion(v)));
218 
219  out.template head<3>() = M1.translation();
220  res_quat = M1.rotation();
221  // Norm of qs might be epsilon-different to 1, so M1.rotation might be epsilon-different to a rotation matrix.
222  // It is then safer to re-normalized after converting M1.rotation to quaternion.
223  firstOrderNormalize(res_quat);
224  }
