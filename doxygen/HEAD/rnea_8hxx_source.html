53 
54  data.v[i] = jdata.v();

58  data.a_gf[i] += data.liMi[i].actInv(data.a_gf[parent]);

87  inline const Eigen::VectorXd&
88  rnea(const Model & model, Data& data,
89  const Eigen::VectorXd & q,
90  const Eigen::VectorXd & v,
91  const Eigen::VectorXd & a)
92  {
93  data.v[0].setZero();
94  data.a_gf[0] = -model.gravity;
95 
96  for( Model::JointIndex i=1;i<(Model::JointIndex)model.njoints;++i )
97  {
98  RneaForwardStep::run(model.joints[i],data.joints[i],
99  RneaForwardStep::ArgsType(model,data,q,v,a));
100  }
101 
102  for( Model::JointIndex i=(Model::JointIndex)model.njoints-1;i>0;--i )
103  {
104  RneaBackwardStep::run(model.joints[i],data.joints[i],
105  RneaBackwardStep::ArgsType(model,data));
106  }
107 
108  return data.tau;
109  }
