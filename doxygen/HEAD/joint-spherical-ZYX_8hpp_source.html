350  void calc (JointDataDerived & data,
351  const Eigen::VectorXd & qs,
352  const Eigen::VectorXd & vs ) const
353  {
354  Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & q = qs.segment<NQ> (idx_q ());
355  Eigen::VectorXd::ConstFixedSegmentReturnType<NV>::Type & q_dot = vs.segment<NQ> (idx_v ());
356 
357  double c0,s0; SINCOS (q(0), &s0, &c0);
358  double c1,s1; SINCOS (q(1), &s1, &c1);
359  double c2,s2; SINCOS (q(2), &s2, &c2);
360 
361  data.M.rotation () << c0 * c1,
362  c0 * s1 * s2 - s0 * c2,
363  c0 * s1 * c2 + s0 * s2,
364  s0 * c1,
365  s0 * s1 * s2 + c0 * c2,
366  s0 * s1 * c2 - c0 * s2,
367  -s1,
368  c1 * s2,
369  c1 * c2;
370 
371 
372  data.S.matrix () << -s1, 0., 1., c1 * s2, c2, 0, c1 * c2, -s2, 0;
373 
374  data.v () = data.S.matrix () * q_dot;
375 
376  data.c ()(0) = -c1 * q_dot (0) * q_dot (1);
377  data.c ()(1) = -s1 * s2 * q_dot (0) * q_dot (1) + c1 * c2 * q_dot (0) * q_dot (2) - s2 * q_dot (1) * q_dot (2);
378  data.c ()(2) = -s1 * c2 * q_dot (0) * q_dot (1) - c1 * s2 * q_dot (0) * q_dot (2) - c2 * q_dot (1) * q_dot (2);
