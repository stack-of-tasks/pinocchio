40  exp3(const Eigen::MatrixBase<D> & v)
41  {
42  EIGEN_STATIC_ASSERT_VECTOR_SPECIFIC_SIZE(D,3);
43  typename D::Scalar nv = v.norm();

45  return Eigen::AngleAxis<typename D::Scalar>(nv, v / nv).matrix();

58  template <typename D> Eigen::Matrix<typename D::Scalar,3,1,Eigen::internal::traits<D>::Options>
59  log3(const Eigen::MatrixBase<D> & R)
60  {
61  EIGEN_STATIC_ASSERT_SAME_MATRIX_SIZE(D, Eigen::Matrix3d);
62  Eigen::AngleAxis<typename D::Scalar> angleAxis(R);
63  assert(0 <= angleAxis.angle() && angleAxis.angle() <= 2 * M_PI);
64  if (angleAxis.angle() > M_PI)
65  return -(2*M_PI - angleAxis.angle()) * angleAxis.axis();
66  return angleAxis.axis() * angleAxis.angle();
67  }

77  template <typename _Scalar, int _Options> SE3Tpl<_Scalar, _Options>
78  exp6(const MotionTpl<_Scalar,_Options> & nu)
79  {
80  typedef _Scalar Scalar;
81  typedef typename MotionTpl<Scalar,_Options>::Vector3 Vector3;
82  typedef typename MotionTpl<Scalar,_Options>::Matrix3 Matrix3;
83 
84  const Vector3 & w = nu.angular();
85  const Vector3 & v = nu.linear();
86  Scalar t = w.norm();
87  if (t > 1e-15)
88  {
89  const double inv_t = 1./t;
90  Matrix3 S(alphaSkew(inv_t, w));
91  double ct,st; SINCOS (t,&st,&ct);
92  Matrix3 V((1. - ct) * inv_t * S + (1. - st * inv_t) * S * S);
93  Vector3 p(v + V * v);
94  return SE3Tpl<_Scalar, _Options>(exp3(w), p);
95  }
96  else
97  {
98  return SE3Tpl<_Scalar, _Options>(Matrix3::Identity(), v);
99  }
100  }

126  template <typename _Scalar, int _Options>
127  MotionTpl<_Scalar,_Options>
128  log6(const SE3Tpl<_Scalar, _Options> & M)
129  {
130  typedef _Scalar Scalar;
131  typedef typename SE3Tpl<Scalar,_Options>::Vector3 Vector3;
132  typedef typename SE3Tpl<Scalar,_Options>::Matrix3 Matrix3;
133 
134  const Matrix3 & R = M.rotation();
135  const Vector3 & p = M.translation();
136  Vector3 w(log3(R));
137  Scalar t = w.norm();
138  if (t > 1e-15)
139  {
140  Matrix3 S(alphaSkew(1./t, w));
141  double ct,st; SINCOS (t,&st,&ct);
142  Matrix3 V(Matrix3::Identity() + (1. - ct)/t * S + (1. - st/t) * S * S);
143 
144  return MotionTpl<_Scalar,_Options>(V.inverse() * p, w);
145  }
146  else
147  return MotionTpl<_Scalar,_Options>(p, w);
148 
149  }
