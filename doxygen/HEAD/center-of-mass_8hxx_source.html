26  inline const SE3::Vector3 &
27  centerOfMass(const Model & model, Data & data,
28  const Eigen::VectorXd & q,
29  const bool computeSubtreeComs,
30  const bool updateKinematics)
31  {
32  data.mass[0] = 0;
33  data.com[0].setZero ();
34 
35  // Forward Step
36  if (updateKinematics)
37  forwardKinematics(model, data, q);
38 
39  for(Model::JointIndex i=1;i<(Model::JointIndex)(model.njoints);++i)
40  {
41  const double mass = model.inertias[i].mass();
42  const SE3::Vector3 & lever = model.inertias[i].lever();
43 
44  data.com[i] = mass * lever;
45  data.mass[i] = mass;
46  }
47 
48  // Backward Step
49  for(Model::JointIndex i=(Model::JointIndex)(model.njoints-1); i>0; --i)
50  {
51  const Model::JointIndex & parent = model.parents[i];
52 
53  const SE3 & liMi = data.liMi[i];
54 
55  data.com[parent] += (liMi.rotation()*data.com[i]
56  + data.mass[i] * liMi.translation());
57  data.mass[parent] += data.mass[i];
58 
59  if(computeSubtreeComs)
60  {
61  data.com[i] /= data.mass[i];
62  }
63  }
64 
65  data.com[0] /= data.mass[0];
66 
67  return data.com[0];
68  }

72  const Eigen::VectorXd & q,

74  const bool computeSubtreeComs,
75  const bool updateKinematics)

80  data.com[0].setZero ();
81  data.vcom[0].setZero ();

83  // Forward Step
84  if (updateKinematics)
85  forwardKinematics(model, data, q, v);
86 
87  for(Model::JointIndex i=1;i<(Model::JointIndex)(model.njoints);++i)
88  {

93 
94  data.com[i] = mass * lever;
95  data.mass[i] = mass;

98  }
99

106

108  + data.mass[i] * liMi.translation());
109 
110  data.vcom[parent] += liMi.rotation()*data.vcom[i];
111  data.mass[parent] += data.mass[i];
112 
113  if( computeSubtreeComs )
114  {
115  data.com[i] /= data.mass[i];
116  data.vcom[i] /= data.mass[i];
117  }
118  }

121  data.vcom[0] /= data.mass[0];

126  inline const SE3::Vector3 &
127  centerOfMass(const Model & model, Data & data,
128  const Eigen::VectorXd & q,
129  const Eigen::VectorXd & v,
130  const Eigen::VectorXd & a,
131  const bool computeSubtreeComs,
132  const bool updateKinematics)
133  {
134  using namespace se3;
135 
136  data.mass[0] = 0;
137  data.com[0].setZero ();
138  data.vcom[0].setZero ();
139  data.acom[0].setZero ();
140 
141  // Forward Step
142  if (updateKinematics)
143  forwardKinematics(model, data, q, v, a);
144 
145  for(Model::JointIndex i=1;i<(Model::JointIndex)(model.njoints);++i)
146  {
147  const double mass = model.inertias[i].mass();
148  const SE3::Vector3 & lever = model.inertias[i].lever();
149 
150  const Motion & v = data.v[i];
151  const Motion & a = data.a[i];
152 
153  data.com[i] = mass * lever;
154  data.mass[i] = mass;
155 
156  data.vcom[i] = mass * (v.angular().cross(lever) + v.linear());
157  data.acom[i] = mass * (a.angular().cross(lever) + a.linear()) + v.angular().cross(data.vcom[i]); // take into accound the coriolis part of the acceleration
158  }
159 
160  // Backward Step
161  for(Model::JointIndex i=(Model::JointIndex)(model.njoints-1); i>0; --i)
162  {
163  const Model::JointIndex & parent = model.parents[i];
164 
165  const SE3 & liMi = data.liMi[i];
166 
167  data.com[parent] += (liMi.rotation()*data.com[i]
168  + data.mass[i] * liMi.translation());
169 
170  data.vcom[parent] += liMi.rotation()*data.vcom[i];
171  data.acom[parent] += liMi.rotation()*data.acom[i];
172  data.mass[parent] += data.mass[i];
173 
174  if( computeSubtreeComs )
175  {
176  data.com[i] /= data.mass[i];
177  data.vcom[i] /= data.mass[i];
178  data.acom[i] /= data.mass[i];
179  }
180  }
181 
182  data.com[0] /= data.mass[0];
183  data.vcom[0] /= data.mass[0];
184  data.acom[0] /= data.mass[0];
185 
186  return data.com[0];
187  }

199  struct JacobianCenterOfMassBackwardStep
200  : public fusion::JointVisitor<JacobianCenterOfMassBackwardStep>
201  {
202  typedef boost::fusion::vector<const se3::Model &,
203  se3::Data &,
204  const bool
205  > ArgsType;

207  JOINT_VISITOR_INIT(JacobianCenterOfMassBackwardStep);

209  template<typename JointModel>
210  static void algo(const se3::JointModelBase<JointModel> & jmodel,
211  se3::JointDataBase<typename JointModel::JointDataDerived> & jdata,
212  const se3::Model& model,
213  se3::Data& data,
214  const bool computeSubtreeComs )
215  {
216  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
217  const Model::JointIndex & parent = model.parents[i];
218 
219  data.com[parent] += data.com[i];
220  data.mass[parent] += data.mass[i];
221 
222  typedef Data::Matrix6x Matrix6x;
223  typedef typename SizeDepType<JointModel::NV>::template ColsReturn<Matrix6x>::Type ColBlock;
224 
225  ColBlock Jcols = jmodel.jointCols(data.J);
226  Jcols = data.oMi[i].act(jdata.S());
227 
228  if (JointModel::NV == -1)
229  {
230  if( jmodel.nv() ==1 )
231  {
232  data.Jcom.col(jmodel.idx_v())
233  = data.mass[i] * Jcols.template topLeftCorner<3,1>()
234  - data.com[i].cross(Jcols.template bottomLeftCorner<3,1>()) ;
235  }
236  else
237  {
238  jmodel.jointCols(data.Jcom)
239  = data.mass[i] * Jcols.template topRows<3>()
240  - skew(data.com[i]) * Jcols.template bottomRows<3>();
241  }
242  }
243  else
244  {
245  if( JointModel::NV ==1 )
246  {
247  data.Jcom.col(jmodel.idx_v())
248  = data.mass[i] * Jcols.template topLeftCorner<3,1>()
249  - data.com[i].cross(Jcols.template bottomLeftCorner<3,1>()) ;
250  }
251  else
252  {
253  jmodel.jointCols(data.Jcom)
254  = data.mass[i] * Jcols.template topRows<3>()
255  - skew(data.com[i]) * Jcols.template bottomRows<3>();
256  }
257  }
258 
259 
260  if(computeSubtreeComs)
261  data.com[i] /= data.mass[i];
262  }
263 
264  };

266  inline const Data::Matrix3x &
267  jacobianCenterOfMass(const Model & model, Data & data,
268  const Eigen::VectorXd & q,
269  const bool computeSubtreeComs,
270  const bool updateKinematics)
271  {
272  data.com[0].setZero ();
273  data.mass[0] = 0;
274 
275  // Forward step
276  if (updateKinematics)
277  forwardKinematics(model, data, q);
278 
279  for(Model::JointIndex i=1;i<(Model::JointIndex)(model.njoints);++i)
280  {
281  const double mass = model.inertias[i].mass();
282  const SE3::Vector3 & lever = model.inertias[i].lever();
283 
284  data.mass[i] = mass;
285  data.com[i] = mass*data.oMi[i].act(lever);
286  }
287 
288  // Backward step
289  for( Model::JointIndex i= (Model::JointIndex) (model.njoints-1);i>0;--i )
290  {
291  JacobianCenterOfMassBackwardStep
292  ::run(model.joints[i],data.joints[i],
293  JacobianCenterOfMassBackwardStep::ArgsType(model,data,computeSubtreeComs));
294  }
295 
296  data.com[0] /= data.mass[0];
297  data.Jcom /= data.mass[0];
298 
299  return data.Jcom;
300  }

302  inline const Data::Matrix3x &
303  getJacobianComFromCrba(const Model & model, Data & data)
304  {
305  const SE3 & oM1 = data.liMi[1];
306 
307  // As the 6 first rows of M*a are a wrench, we just need to multiply by the
308  // relative rotation between the first joint and the world
309  const SE3::Matrix3 oR1_over_m (oM1.rotation() / data.M(0,0));
310 
311  // I don't know why, but the colwise multiplication is much more faster
312  // than the direct Eigen multiplication
313  for (long k=0; k<model.nv;++k)
314  data.Jcom.col(k) = oR1_over_m * data.M.topRows<3> ().col(k);
315  return data.Jcom;
316  }
