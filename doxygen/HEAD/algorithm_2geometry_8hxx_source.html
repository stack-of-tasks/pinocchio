61  inline bool computeCollision(const GeometryModel & geomModel,
62  GeometryData & geomData,
63  const PairIndex& pairId)
64  {
65  assert( pairId < geomModel.collisionPairs.size() );
66  const CollisionPair & pair = geomModel.collisionPairs[pairId];
67 
68  assert( pairId < geomData.collisionResults.size() );
69  assert( pair.first < geomData.collisionObjects.size() );
70  assert( pair.second < geomData.collisionObjects.size() );
71 
72  fcl::CollisionResult& collisionResult = geomData.collisionResults[pairId];
73  collisionResult.clear();
74  fcl::collide (&geomData.collisionObjects[pair.first],
75  &geomData.collisionObjects[pair.second],
76  geomData.collisionRequest,
77  collisionResult);
78 
79  return collisionResult.isCollision();
80  }

82  inline bool computeCollisions(const GeometryModel & geomModel,
83  GeometryData & geomData,
84  const bool stopAtFirstCollision = true)
85  {
86  bool isColliding = false;
87 
88  for (std::size_t cpt = 0; cpt < geomModel.collisionPairs.size(); ++cpt)
89  {
90  if(geomData.activeCollisionPairs[cpt])
91  {
92  computeCollision(geomModel,geomData,cpt);
93  isColliding |= geomData.collisionResults[cpt].isCollision();
94  if(isColliding && stopAtFirstCollision)
95  return true;
96  }
97  }
98 
99  return isColliding;
100  }

141  template <bool COMPUTE_SHORTEST>
142  inline std::size_t computeDistances(const GeometryModel & geomModel,
143  GeometryData & geomData)
144  {
145  std::size_t min_index = geomModel.collisionPairs.size();
146  double min_dist = std::numeric_limits<double>::infinity();
147  for (std::size_t cpt = 0; cpt < geomModel.collisionPairs.size(); ++cpt)
148  {
149  if(geomData.activeCollisionPairs[cpt])
150  {
151  computeDistance(geomModel,geomData,cpt);
152  if (COMPUTE_SHORTEST && geomData.distanceResults[cpt].min_distance < min_dist)
153  {
154  min_index = cpt;
155  min_dist = geomData.distanceResults[cpt].min_distance;
156  }
157  }
158  }
159  return min_index;
160  }

207  inline void computeBodyRadius(const Model & model,
208  const GeometryModel & geomModel,
209  GeometryData & geomData)
210  {
211  geomData.radius.resize(model.joints.size(),0);
212  BOOST_FOREACH(const GeometryObject & geom,geomModel.geometryObjects)
213  {
214  const boost::shared_ptr<const fcl::CollisionGeometry> & fcl
215  = geom.fcl;
216  const SE3 & jMb = geom.placement; // placement in joint.
217  const Model::JointIndex & i = geom.parentJoint;
218  assert (i<geomData.radius.size());
219 
220  double radius = geomData.radius[i] * geomData.radius[i];
221 
222  // The radius is simply the one of the 8 corners of the AABB cube, expressed
223  // in the joint frame, whose norm is the highest.
224  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,min,min,min)).squaredNorm(),radius);
225  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,min,min,max)).squaredNorm(),radius);
226  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,min,max,min)).squaredNorm(),radius);
227  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,min,max,max)).squaredNorm(),radius);
228  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,max,min,min)).squaredNorm(),radius);
229  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,max,min,max)).squaredNorm(),radius);
230  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,max,max,min)).squaredNorm(),radius);
231  radius = std::max (jMb.act(SE3_GEOM_AABB(fcl,max,max,max)).squaredNorm(),radius);
232 
233  // Don't forget to sqroot the squared norm before storing it.
234  geomData.radius[i] = sqrt(radius);
235  }
236  }

243  inline void appendGeometryModel(GeometryModel & geomModel1,
244  const GeometryModel & geomModel2)
245  {
246  assert (geomModel1.ngeoms == geomModel1.geometryObjects.size());
247  Index nGeom1 = geomModel1.geometryObjects.size();
248  Index nColPairs1 = geomModel1.collisionPairs.size();
249  assert (geomModel2.ngeoms == geomModel2.geometryObjects.size());
250  Index nGeom2 = geomModel2.geometryObjects.size();
251  Index nColPairs2 = geomModel2.collisionPairs.size();
252 
254  geomModel1.geometryObjects.insert(geomModel1.geometryObjects.end(),
255  geomModel2.geometryObjects.begin(), geomModel2.geometryObjects.end());
256  geomModel1.ngeoms += nGeom2;
257 
259  geomModel1.collisionPairs.reserve(nColPairs1 + nColPairs2 + nGeom1 * nGeom2);
260  for (Index i = 0; i < nColPairs2; ++i)
261  {
262  const CollisionPair& cp = geomModel2.collisionPairs[i];
263  geomModel1.collisionPairs.push_back(
264  CollisionPair (cp.first + nGeom1, cp.second + nGeom1)
265  );
266  }
267 
269  for (Index i = 0; i < nGeom1; ++i) {
270  for (Index j = 0; j < nGeom2; ++j) {
271  geomModel1.collisionPairs.push_back(CollisionPair(i, nGeom1 + j));
272  }
273  }
274  }
