130  InertiaTpl(const Matrix6 & I6)
131  {
132  assert((I6 - I6.transpose()).isMuchSmallerThan(I6));
133  m = I6(LINEAR, LINEAR);
134  const Matrix3 & mc_cross = I6.template block <3,3> (ANGULAR,LINEAR);
135  c = unSkew(mc_cross);
136  c /= m;
137 
138  Matrix3 I3 (mc_cross * mc_cross);
139  I3 /= m;
140  I3 += I6.template block<3,3>(ANGULAR,ANGULAR);
141  I = Symmetric3(I3);
142  }

230  Matrix6 matrix_impl() const
231  {
232  Matrix6 M;
233  const Matrix3 & c_cross = (skew(c));
234  M.template block<3,3>(LINEAR, LINEAR ).setZero ();
235  M.template block<3,3>(LINEAR, LINEAR ).diagonal ().fill (m);
236  M.template block<3,3>(ANGULAR,LINEAR ) = m * c_cross;
237  M.template block<3,3>(LINEAR, ANGULAR) = -M.template block<3,3> (ANGULAR, LINEAR);
238  M.template block<3,3>(ANGULAR,ANGULAR) = I - M.template block<3,3>(ANGULAR, LINEAR) * c_cross;
239 
240  return M;
241  }

264  InertiaTpl __plus__(const InertiaTpl &Yb) const
265  {
266  /* Y_{a+b} = ( m_a+m_b,
267  * (m_a*c_a + m_b*c_b ) / (m_a + m_b),
268  * I_a + I_b - (m_a*m_b)/(m_a+m_b) * AB_x * AB_x )
269  */
270 
271  const double & mab = m+Yb.m;
272  const Vector3 & AB = (c-Yb.c).eval();
273  return InertiaTpl( mab,
274  (m*c+Yb.m*Yb.c)/mab,
275  I+Yb.I - (m*Yb.m/mab)* typename Symmetric3::SkewSquare(AB));
276  }
277 
278  InertiaTpl& __pequ__(const InertiaTpl &Yb)
279  {
280  const InertiaTpl& Ya = *this;
281  const double & mab = Ya.m+Yb.m;
282  const Vector3 & AB = (Ya.c-Yb.c).eval();
283  c *= m; c += Yb.m*Yb.c; c /= mab;
284  I += Yb.I; I -= (Ya.m*Yb.m/mab)* typename Symmetric3::SkewSquare(AB);
285  m = mab;
286  return *this;
287  }

289  Force __mult__(const Motion &v) const
290  {
291  Force f;
292  f.linear() = m*(v.linear() - c.cross(v.angular()));
293  f.angular() = c.cross(f.linear()) + I*v.angular();
294  return f;
295  }

297  Scalar vtiv_impl(const Motion & v) const
298  {
299  const Vector3 cxw (c.cross(v.angular()));
300  Scalar res = m * (v.linear().squaredNorm() - 2.*v.linear().dot(cxw));
301  const Vector3 mcxcxw (-m*c.cross(cxw));
302  res += v.angular().dot(mcxcxw);
303  res += I.vtiv(v.angular());
304 
305  return res;
306  }
