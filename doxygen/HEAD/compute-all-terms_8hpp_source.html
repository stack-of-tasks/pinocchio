69  template<typename JointModel>
70  static void algo(const se3::JointModelBase<JointModel> & jmodel,
71  se3::JointDataBase<typename JointModel::JointDataDerived> & jdata,
72  const se3::Model & model,
73  se3::Data & data,
74  const Eigen::VectorXd & q,
75  const Eigen::VectorXd & v)
76  {
77  using namespace Eigen;
78  using namespace se3;
79 
80  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
81  const Model::JointIndex & parent = model.parents[i];
82 
83  jmodel.calc(jdata.derived(),q,v);
84 
85  // CRBA
86  data.liMi[i] = model.jointPlacements[i]*jdata.M();
87  data.Ycrb[i] = model.inertias[i];
88 
89  // Jacobian + NLE
90  data.v[i] = jdata.v();
91 
92  if(parent>0)
93  {
94  data.oMi[i] = data.oMi[parent]*data.liMi[i];
95  data.v[i] += data.liMi[i].actInv(data.v[parent]);
96  }
97  else
98  data.oMi[i] = data.liMi[i];
99 
100  jmodel.jointCols(data.J) = data.oMi[i].act(jdata.S());
101 
102  data.a_gf[i] = data.a[i] = jdata.c() + (data.v[i] ^ jdata.v());
103  if (parent > 0)
104  data.a[i] += data.liMi[i].actInv(data.a[parent]);
105 
106  data.a_gf[i] += data.liMi[i].actInv(data.a_gf[parent]);
107 
108  data.f[i] = model.inertias[i]*data.a_gf[i] + model.inertias[i].vxiv(data.v[i]); // -f_ext
109 
110  // CoM
111  const double mass = model.inertias[i].mass();
112  const SE3::Vector3 & lever = model.inertias[i].lever();
113 
114  data.com[i] = mass * lever;
115  data.mass[i] = mass;
116 
117  data.vcom[i] = mass * (data.v[i].angular().cross(lever) + data.v[i].linear());
118  }
119 
120  };

129  template<typename JointModel>
130  static void algo(const JointModelBase<JointModel> & jmodel,
131  JointDataBase<typename JointModel::JointDataDerived> & jdata,
132  const Model & model,
133  Data & data)
134  {
135  /*
136  * F[1:6,i] = Y*S
137  * M[i,SUBTREE] = S'*F[1:6,SUBTREE]
138  * if li>0
139  * Yli += liXi Yi
140  * F[1:6,SUBTREE] = liXi F[1:6,SUBTREE]
141  */
142  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
143  const Model::JointIndex & parent = model.parents[i];
144  const SE3 & oMi = data.oMi[i];
145 
146  /* F[1:6,i] = Y*S */
147  jmodel.jointCols(data.Fcrb[i]) = data.Ycrb[i] * jdata.S();
148 
149  /* M[i,SUBTREE] = S'*F[1:6,SUBTREE] */
150  data.M.block(jmodel.idx_v(),jmodel.idx_v(),jmodel.nv(),data.nvSubtree[i])
151  = jdata.S().transpose()*data.Fcrb[i].middleCols(jmodel.idx_v(),data.nvSubtree[i]);
152 
153 
154  jmodel.jointVelocitySelector(data.nle) = jdata.S().transpose()*data.f[i];
155  if(parent>0)
156  {
157  /* Yli += liXi Yi */
158  data.Ycrb[parent] += data.liMi[i].act(data.Ycrb[i]);
159 
160  /* F[1:6,SUBTREE] = liXi F[1:6,SUBTREE] */
161  Eigen::Block<typename Data::Matrix6x> jF
162  = data.Fcrb[parent].block(0,jmodel.idx_v(),6,data.nvSubtree[i]);
163  Eigen::Block<typename Data::Matrix6x> iF
164  = data.Fcrb[i].block(0,jmodel.idx_v(),6,data.nvSubtree[i]);
165  forceSet::se3Action(data.liMi[i], iF, jF);
166 
167  data.f[parent] += data.liMi[i].act(data.f[i]);
168  }
169 
170  // CoM
171  const SE3 & liMi = data.liMi[i];
172 
173  data.com[parent] += (liMi.rotation()*data.com[i]
174  + data.mass[i] * liMi.translation());
175 
176  SE3::Vector3 com_in_world (oMi.rotation() * data.com[i] + data.mass[i] * oMi.translation());
177 
178  data.vcom[parent] += liMi.rotation()*data.vcom[i];
179  data.mass[parent] += data.mass[i];
180 
181  typedef Data::Matrix6x Matrix6x;
182  typedef typename SizeDepType<JointModel::NV>::template ColsReturn<Matrix6x>::Type ColBlock;
183 
184  ColBlock Jcols = jmodel.jointCols(data.J);
185 
186  if( JointModel::NV==1 )
187  data.Jcom.col(jmodel.idx_v())
188  = data.mass[i] * Jcols.template topLeftCorner<3,1>()
189  - com_in_world.cross(Jcols.template bottomLeftCorner<3,1>()) ;
190  else
191  jmodel.jointCols(data.Jcom)
192  = data.mass[i] * Jcols.template topRows<3>()
193  - skew(com_in_world) * Jcols.template bottomRows<3>();
194 
195  data.com[i] /= data.mass[i];
196  data.vcom[i] /= data.mass[i];
197  }
198  };

200  inline void
201  computeAllTerms(const Model & model,
202  Data & data,
203  const Eigen::VectorXd & q,
204  const Eigen::VectorXd & v)
205  {
206  data.v[0].setZero();
207  data.a[0].setZero();
208  data.a_gf[0] = -model.gravity;
209 
210  data.mass[0] = 0;
211  data.com[0].setZero ();
212  data.vcom[0].setZero ();
213 
214  for(Model::JointIndex i=1;i<(Model::JointIndex) model.njoints;++i)
215  {
216  CATForwardStep::run(model.joints[i],data.joints[i],
217  CATForwardStep::ArgsType(model,data,q,v));
218  }
219 
220  for(Model::JointIndex i=(Model::JointIndex)(model.njoints-1);i>0;--i)
221  {
222  CATBackwardStep::run(model.joints[i],data.joints[i],
223  CATBackwardStep::ArgsType(model,data));
224  }
225 
226  // CoM
227  data.com[0] /= data.mass[0];
228  data.vcom[0] /= data.mass[0];
229 
230  // JCoM
231  data.Jcom /= data.mass[0];
232 
233  // Energy
234  kineticEnergy(model, data, q, v, false);
235  potentialEnergy(model, data, q, false);
236 
237  }
