49  inline const Eigen::VectorXd & forwardDynamics(const Model & model,
50  Data & data,
51  const Eigen::VectorXd & q,
52  const Eigen::VectorXd & v,
53  const Eigen::VectorXd & tau,
54  const Eigen::MatrixXd & J,
55  const Eigen::VectorXd & gamma,
56  const bool updateKinematics = true
57  )
58  {
59  assert(q.size() == model.nq);
60  assert(v.size() == model.nv);
61  assert(tau.size() == model.nv);
62  assert(J.cols() == model.nv);
63  assert(J.rows() == gamma.size());
64 
65  Eigen::VectorXd & a = data.ddq;
66  Eigen::VectorXd & lambda_c = data.lambda_c;
67 
68  if (updateKinematics)
69  computeAllTerms(model, data, q, v);
70 
71  // Compute the UDUt decomposition of data.M
72  cholesky::decompose(model, data);
73 
74  // Compute the dynamic drift (control - nle)
75  data.torque_residual = tau - data.nle;
76  cholesky::solve(model, data, data.torque_residual);
77 
78  data.sDUiJt = J.transpose();
79  // Compute U^-1 * J.T
80  cholesky::Uiv(model, data, data.sDUiJt);
81  for(int k=0;k<model.nv;++k) data.sDUiJt.row(k) /= sqrt(data.D[k]);
82 
83  data.JMinvJt.noalias() = data.sDUiJt.transpose() * data.sDUiJt;
84  data.llt_JMinvJt.compute(data.JMinvJt);
85 
86  // Compute the Lagrange Multipliers
87  lambda_c = -gamma -J*data.torque_residual;
88  data.llt_JMinvJt.solveInPlace (lambda_c);
89 
90  // Compute the joint acceleration
91  a = J.transpose() * lambda_c;
92  cholesky::solve (model, data, a);
93  a += data.torque_residual;
94 
95  return a;
96  }

116  inline const Eigen::VectorXd & impulseDynamics(const Model & model,
117  Data & data,
118  const Eigen::VectorXd & q,
119  const Eigen::VectorXd & v_before,
120  const Eigen::MatrixXd & J,
121  const double r_coeff = 0.,
122  const bool updateKinematics = true
123  )
124  {
125  assert(q.size() == model.nq);
126  assert(v_before.size() == model.nv);
127  assert(J.cols() == model.nv);
128 
129  Eigen::VectorXd & impulse_c = data.impulse_c;
130  Eigen::VectorXd & dq_after = data.dq_after;
131 
132  // Compute the mass matrix
133  if (updateKinematics)
134  crba(model, data, q);
135 
136  // Compute the UDUt decomposition of data.M
137  cholesky::decompose(model, data);
138 
139  data.sDUiJt = J.transpose();
140  // Compute U^-1 * J.T
141  cholesky::Uiv(model, data, data.sDUiJt);
142  for(int k=0;k<model.nv;++k) data.sDUiJt.row(k) /= sqrt(data.D[k]);
143 
144  data.JMinvJt.noalias() = data.sDUiJt.transpose() * data.sDUiJt;
145  data.llt_JMinvJt.compute(data.JMinvJt);
146 
147  // Compute the Lagrange Multipliers related to the contact impulses
148  impulse_c = (-r_coeff - 1.) * (J * v_before);
149  data.llt_JMinvJt.solveInPlace (impulse_c);
150 
151  // Compute the joint velocity after impacts
152  dq_after = J.transpose() * impulse_c;
153  cholesky::solve (model, data, dq_after);
154  dq_after += v_before;
155 
156  return dq_after;
157  }
