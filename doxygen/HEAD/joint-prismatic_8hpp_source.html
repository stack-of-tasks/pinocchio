211  inline Motion operator^( const Motion& m1, const MotionPrismatic<0>& m2)
212  {
213  /* nu1^nu2 = ( v1^w2+w1^v2, w1^w2 )
214  * nu1^(v2,0) = ( w1^v2 , 0 )
215  * (x,y,z)^(v,0,0) = ( 0,zv,-yv )
216  * nu1^(0,vx) = ( 0,wz1 vx,-wy1 vx, 0, 0, 0)
217  */
218  const Motion::Vector3& w = m1.angular();
219  const double & vx = m2.v;
220  return Motion( Motion::Vector3(0,w[2]*vx,-w[1]*vx),
221  Motion::Vector3::Zero());
222  }
223 
224  inline Motion operator^( const Motion& m1, const MotionPrismatic<1>& m2)
225  {
226  /* nu1^nu2 = ( v1^w2+w1^v2, w1^w2 )
227  * nu1^(v2,0) = ( w1^v2 , 0 )
228  * (x,y,z)^(0,v,0) = ( -zv,0,xv )
229  * nu1^(0,vx) = ( -vz1 vx,0,vx1 vx, 0, 0, 0)
230  */
231  const Motion::Vector3& w = m1.angular();
232  const double & vx = m2.v;
233  return Motion( Motion::Vector3(-w[2]*vx,0,w[0]*vx),
234  Motion::Vector3::Zero());
235  }
236 
237  inline Motion operator^( const Motion& m1, const MotionPrismatic<2>& m2)
238  {
239  /* nu1^nu2 = ( v1^w2+w1^v2, w1^w2 )
240  * nu1^(v2,0) = ( w1^v2 , 0 )
241  * (x,y,z)^(0,0,v) = ( yv,-xv,0 )
242  * nu1^(0,vx) = ( vy1 vx,-vx1 vx, 0, 0, 0, 0 )
243  */
244  const Motion::Vector3& w = m1.angular();
245  const double & vx = m2.v;
