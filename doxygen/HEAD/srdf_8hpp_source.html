37  inline void removeCollisionPairsFromSrdf(const Model& model,
48  GeometryModel & geomModel,
49  const std::string & filename,
50  const bool verbose) throw (std::invalid_argument)
51  {
52  // Check extension
53  const std::string extension = filename.substr(filename.find_last_of('.')+1);
54  if (extension != "srdf")
55  {
56  const std::string exception_message (filename + " does not have the right extension.");
57  throw std::invalid_argument(exception_message);
58  }
59 
60  // Open file
61  std::ifstream srdf_stream(filename.c_str());
62  if (! srdf_stream.is_open())
63  {
64  const std::string exception_message (filename + " does not seem to be a valid file.");
65  throw std::invalid_argument(exception_message);
66  }
67 
68  // Read xml stream
69  using boost::property_tree::ptree;
70  ptree pt;
71  read_xml(srdf_stream, pt);
72 
73  // Iterate over collision pairs
74  BOOST_FOREACH(const ptree::value_type & v, pt.get_child("robot"))
75  {
76  if (v.first == "disable_collisions")
77  {
78  const std::string link1 = v.second.get<std::string>("<xmlattr>.link1");
79  const std::string link2 = v.second.get<std::string>("<xmlattr>.link2");
80 
81  // Check first if the two bodies exist in model
82  if (!model.existBodyName(link1) || !model.existBodyName(link2))
83  {
84  if (verbose)
85  std::cout << "It seems that " << link1 << " or " << link2 << " do not exist in model. Skip." << std::endl;
86  continue;
87  }
88 
89  const Model::JointIndex frame_id1 = model.getBodyId(link1);
90  const Model::JointIndex frame_id2 = model.getBodyId(link2);
91 
92  // Malformed SRDF
93  if (frame_id1 == frame_id2)
94  {
95  if (verbose)
96  std::cout << "Cannot disable collision between " << link1 << " and " << link2 << std::endl;
97  continue;
98  }
99 
100  typedef std::vector<CollisionPair> CollisionPairs_t;
101  bool didRemove = false;
102  for(CollisionPairs_t::iterator _colPair = geomModel.collisionPairs.begin();
103  _colPair != geomModel.collisionPairs.end(); ) {
104  const CollisionPair& colPair (*_colPair);
105  bool remove =
106  (
107  (geomModel.geometryObjects[colPair.first ].parentFrame == frame_id1)
108  && (geomModel.geometryObjects[colPair.second].parentFrame == frame_id2)
109  ) || (
110  (geomModel.geometryObjects[colPair.second].parentFrame == frame_id1)
111  && (geomModel.geometryObjects[colPair.first ].parentFrame == frame_id2)
112  );
113 
114  if (remove) {
115  _colPair = geomModel.collisionPairs.erase(_colPair);
116  didRemove = true;
117  } else {
118  ++_colPair;
119  }
120  }
121  if(didRemove && verbose)
122  std::cout << "Remove collision pair (" << link1 << "," << link2 << ")" << std::endl;
123 
124  }
125  } // BOOST_FOREACH
126  }

140  inline Eigen::VectorXd getNeutralConfigurationFromSrdf(Model & model,
141  const std::string & filename,
142  const bool verbose) throw (std::invalid_argument)
143  {
144  // Check extension
145  const std::string extension = filename.substr(filename.find_last_of('.')+1);
146  if (extension != "srdf")
147  {
148  const std::string exception_message (filename + " does not have the right extension.");
149  throw std::invalid_argument(exception_message);
150  }
151 
152  // Open file
153  std::ifstream srdf_stream(filename.c_str());
154  if (! srdf_stream.is_open())
155  {
156  const std::string exception_message (filename + " does not seem to be a valid file.");
157  throw std::invalid_argument(exception_message);
158  }
159 
160  // Read xml stream
161  using boost::property_tree::ptree;
162  ptree pt;
163  read_xml(srdf_stream, pt);
164 
165  // Iterate over all tags directly children of robot
166  BOOST_FOREACH(const ptree::value_type & v, pt.get_child("robot"))
167  {
168  // if we encounter a tag group_state
169  if (v.first == "group_state")
170  {
171  const std::string name = v.second.get<std::string>("<xmlattr>.name");
172  // Ensure that it is the half_sitting tag
173  if( name == "half_sitting")
174  {
175  // Iterate over all the joint tags
176  BOOST_FOREACH(const ptree::value_type & joint, v.second)
177  {
178  if (joint.first == "joint")
179  {
180  std::string joint_name = joint.second.get<std::string>("<xmlattr>.name");
181  double joint_config = joint.second.get<double>("<xmlattr>.value");
182  if (verbose)
183  {
184  std::cout << "(" << joint_name << " , " << joint_config << ")" << std::endl;
185  }
186  // Search in model the joint and its config id
187  Model::JointIndex joint_id = model.getJointId(joint_name);
188 
189  if (joint_id != model.joints.size()) // != model.njoints
190  {
191  const JointModel & joint = model.joints[joint_id];
192  model.neutralConfiguration(joint.idx_q()) = joint_config; // joint with 1 dof
193  // model.neutralConfiguration.segment(joint.idx_q(),joint.nq()) = joint_config; // joint with more than 1 dof
194  }
195  else
196  {
197  if (verbose) std::cout << "The Joint " << joint_name << " was not found in model" << std::endl;
198  }
199  }
200  }
201  return model.neutralConfiguration;
202  }
203 
204  }
205  } // BOOST_FOREACH
206  assert(false && "no half_sitting configuration found in the srdf file"); // Should we throw something here ?
207  return Eigen::VectorXd::Constant(model.nq,NAN); // warning : uninitialized vector is returned
208  }
209  }
