123  inline static Inertia::Matrix6 SE3actOn(const SE3 & M, const Inertia::Matrix6 & I)
124  {
125  typedef Inertia::Matrix6 Matrix6;
126  typedef SE3::Matrix3 Matrix3;
127  typedef SE3::Vector3 Vector3;
128  typedef Eigen::Block<const Matrix6,3,3> constBlock3;
129  typedef Eigen::Block<Matrix6,3,3> Block3;
130 
131  const constBlock3 & Ai = I.block<3,3> (Inertia::LINEAR, Inertia::LINEAR);
132  const constBlock3 & Bi = I.block<3,3> (Inertia::LINEAR, Inertia::ANGULAR);
133  const constBlock3 & Di = I.block<3,3> (Inertia::ANGULAR, Inertia::ANGULAR);
134 
135  const Matrix3 & R = M.rotation();
136  const Vector3 & t = M.translation();
137 
138  Matrix6 res;
139  Block3 Ao = res.block<3,3> (Inertia::LINEAR, Inertia::LINEAR);
140  Block3 Bo = res.block<3,3> (Inertia::LINEAR, Inertia::ANGULAR);
141  Block3 Co = res.block<3,3> (Inertia::ANGULAR, Inertia::LINEAR);
142  Block3 Do = res.block<3,3> (Inertia::ANGULAR, Inertia::ANGULAR);
143 
144  Ao = R*Ai*R.transpose();
145  Bo = R*Bi*R.transpose();
146  Do.row(0) = t.cross(Bo.col(0));
147  Do.row(1) = t.cross(Bo.col(1));
148  Do.row(2) = t.cross(Bo.col(2));
149 
150  Co.col(0) = t.cross(Ao.col(0));
151  Co.col(1) = t.cross(Ao.col(1));
152  Co.col(2) = t.cross(Ao.col(2));
153  Co += Bo.transpose();
154 
155  Bo = Co.transpose();
156  Do.col(0) += t.cross(Bo.col(0));
157  Do.col(1) += t.cross(Bo.col(1));
158  Do.col(2) += t.cross(Bo.col(2));
159  Do += R*Di*R.transpose();
160  return res;
161  }

189  inline const Eigen::VectorXd &
190  aba(const Model & model,
191  Data & data,
192  const Eigen::VectorXd & q,
193  const Eigen::VectorXd & v,
194  const Eigen::VectorXd & tau)
195  {
196  data.v[0].setZero();
197  data.a[0] = -model.gravity;
198  data.u = tau;
199 
200  for(Model::Index i=1;i<(Model::Index)model.njoints;++i)
201  {
202  AbaForwardStep1::run(model.joints[i],data.joints[i],
203  AbaForwardStep1::ArgsType(model,data,q,v));
204  }
205 
206  for( Model::Index i=(Model::Index)model.njoints-1;i>0;--i )
207  {
208  AbaBackwardStep::run(model.joints[i],data.joints[i],
209  AbaBackwardStep::ArgsType(model,data));
210  }
211 
212  for(Model::Index i=1;i<(Model::Index)model.njoints;++i)
213  {
214  AbaForwardStep2::run(model.joints[i],data.joints[i],
215  AbaForwardStep2::ArgsType(model,data));
216  }
217 
218  return data.ddq;
219  }

232  || (model.inertias[j].inertia().data()[3] < 1e-5)
233  || (model.inertias[j].inertia().data()[5] < 1e-5) )
234  return false;
