194  template <class Derived>
195  template <class ConfigL_t, class ConfigR_t, class ConfigOut_t>
196  void LieGroupOperationBase<Derived>::interpolate_impl(
197  const Eigen::MatrixBase<ConfigL_t> & q0,
198  const Eigen::MatrixBase<ConfigR_t> & q1,
199  const Scalar& u,
200  const Eigen::MatrixBase<ConfigOut_t>& qout)
201  {
202  if (u == 0) const_cast<Eigen::MatrixBase<ConfigOut_t>&>(qout) = q0;
203  else if(u == 1) const_cast<Eigen::MatrixBase<ConfigOut_t>&>(qout) = q1;
204  else integrate(q0, u * difference(q0, q1), qout);
205  }
