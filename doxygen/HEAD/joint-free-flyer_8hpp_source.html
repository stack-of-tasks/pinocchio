257  ConfigVector_t integrate_impl(const Eigen::VectorXd & qs, const Eigen::VectorXd & vs) const
258  {
259  typedef Eigen::Map<Motion_t::Quaternion_t> QuaternionMap_t;
260 
261  Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & q = qs.segment<NQ> (idx_q ());
262  Eigen::VectorXd::ConstFixedSegmentReturnType<NV>::Type & q_dot = vs.segment<NV> (idx_v ());
263 
264  ConfigVector_t res;
265  Transformation_t M0; forwardKinematics(M0,q);
266  Transformation_t M1(M0*exp6(Motion_t(q_dot)));
267 
268  res.head<3>() = M1.translation();
269  QuaternionMap_t res_quat(res.tail<4>().data());
270  res_quat = M1.rotation();
271  // Norm of qs might be epsilon-different to 1, so M1.rotation might be epsilon-different to a rotation matrix.
272  // It is then safer to re-normalized after converting M1.rotation to quaternion.
273  firstOrderNormalize(res_quat);
274 
275  return res;
276  }

278  ConfigVector_t interpolate_impl(const Eigen::VectorXd & q0, const Eigen::VectorXd & q1, const double u) const
279  {
280  typedef Eigen::Map<Motion_t::Quaternion_t> QuaternionMap_t;
281 
282  Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & q_0 = q0.segment<NQ> (idx_q ());
283  Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & q_1 = q1.segment<NQ> (idx_q ());
284 
285  if (u == 0.) return q_0;
286  else if( u == 1.) return q_1;
287  else
288  {
289  // TODO: If integrate takes an arguments (ConfigVector_t, TangentVector_t), then we can merely do:
290  // TangentVector_t nu(u*difference(q0, q1));
291  // return integrate(q0, nu);
292 
293  TangentVector_t nu(u*difference(q0, q1));
294  Transformation_t M0; forwardKinematics(M0,q_0);
295  Transformation_t M1(M0*exp6(Motion_t(nu)));
296 
297  ConfigVector_t res;
298  res.head<3>() = M1.translation();
299  QuaternionMap_t res_quat(res.tail<4>().data());
300  res_quat = M1.rotation();
301  firstOrderNormalize(res_quat);
302 
303  return res;
304  }

317  ConfigVector_t randomConfiguration_impl(const ConfigVector_t & lower_pos_limit, const ConfigVector_t & upper_pos_limit ) const throw (std::runtime_error)
318  {
319  ConfigVector_t result;
320  // Translational part
321  for (Index i = 0; i < 3; ++i)
322  {
323  if(lower_pos_limit[i] == -std::numeric_limits<double>::infinity() ||
324  upper_pos_limit[i] == std::numeric_limits<double>::infinity() )
325  {
326  std::ostringstream error;
327  error << "non bounded limit. Cannot uniformly sample joint nb " << id();
328  throw std::runtime_error(error.str());
329  }
330 
331  result[i] = lower_pos_limit[i] + (upper_pos_limit[i] - lower_pos_limit[i]) * (Scalar)(rand())/RAND_MAX;
332  }
333 
334  typedef Eigen::Map<Motion_t::Quaternion_t> QuaternionMap_t;
335  uniformRandom(QuaternionMap_t(result.segment<4>(3).data()));
336 
337  return result;
338  }
