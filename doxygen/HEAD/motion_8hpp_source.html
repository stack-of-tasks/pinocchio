154  ActionMatrix_t toActionMatrix_impl () const
155  {
156  ActionMatrix_t X;
157  X.block <3,3> (ANGULAR, ANGULAR) = X.block <3,3> (LINEAR, LINEAR) = skew (angular_impl());
158  X.block <3,3> (LINEAR, ANGULAR) = skew (linear_impl());
159  X.block <3,3> (ANGULAR, LINEAR).setZero ();
160 
161  return X;
162  }

168  Linear_t linear_impl() { return data.template segment<3> (LINEAR); }
