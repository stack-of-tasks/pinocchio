290  void calc_aba(JointDataDerived & data, Inertia::Matrix6 & I, const bool update_I) const
291  {
292  data.U = I.block<6,3> (0,Inertia::LINEAR);
293  data.Dinv = I.block<3,3> (Inertia::LINEAR,Inertia::LINEAR).inverse();
294  data.UDinv.middleRows<3> (Inertia::LINEAR).setIdentity(); // can be put in data constructor
295  data.UDinv.middleRows<3> (Inertia::ANGULAR) = data.U.block<3,3> (Inertia::ANGULAR, 0) * data.Dinv;
296 
297  if (update_I)
298  {
299  I.block<6,3> (0,Inertia::LINEAR).setZero();
300  I.block<3,3> (Inertia::LINEAR,Inertia::ANGULAR).setZero();
301  I.block<3,3> (Inertia::ANGULAR,Inertia::ANGULAR) -= data.UDinv.middleRows<3> (Inertia::ANGULAR) * I.block<3,3> (Inertia::LINEAR, Inertia::ANGULAR);
302  }

334  ConfigVector_t randomConfiguration_impl(const ConfigVector_t & lower_pos_limit, const ConfigVector_t & upper_pos_limit ) const throw (std::runtime_error)
335  {
336  ConfigVector_t result;
337  for (int i = 0; i < result.size(); ++i)
338  {
339  if(lower_pos_limit[i] == -std::numeric_limits<double>::infinity() ||
340  upper_pos_limit[i] == std::numeric_limits<double>::infinity() )
341  {
342  std::ostringstream error;
343  error << "non bounded limit. Cannot uniformly sample joint nb " << id() ;
344  assert(false && "non bounded limit. Cannot uniformly sample joint translation");
345  throw std::runtime_error(error.str());
346  }
347  result[i] = lower_pos_limit[i] + ( upper_pos_limit[i] - lower_pos_limit[i]) * rand()/RAND_MAX;
348  }
349  return result;
350  }
