217  /* [CRBA] ForceSet operator* (Inertia Y,Constraint S) */
218  inline Eigen::Matrix <Inertia::Scalar, 6, 3> operator* (const Inertia & Y, const ConstraintPlanar &)
219  {
220  Eigen::Matrix <Inertia::Scalar, 6, 3> M;
221  const double mass = Y.mass ();
222  const Inertia::Vector3 & com = Y.lever ();
223  const Symmetric3 & inertia = Y.inertia ();
224 
225  M.topLeftCorner <3,3> ().setZero ();
226  M.topLeftCorner <2,2> ().diagonal ().fill (mass);
227 
228  Inertia::Vector3 mc (mass * com);
229  M.rightCols <1> ().head <2> () << -mc(1), mc(0);
230 
231  M.bottomLeftCorner <3,2> () << 0., -mc(2), mc(2), 0., -mc(1), mc(0);
232  M.rightCols <1> ().tail <3> () = inertia.data ().tail <3> ();
233  M.rightCols <1> ()[3] -= mc(0)*com(2);
234  M.rightCols <1> ()[4] -= mc(1)*com(2);
235  M.rightCols <1> ()[5] += mass*(com(0)*com(0) + com(1)*com(1));
236 
237  return M;
238  }

389  ConfigVector_t integrate_impl(const Eigen::VectorXd & qs,const Eigen::VectorXd & vs) const
390  {
391  Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & q = qs.segment<NQ> (idx_q ());
392  Eigen::VectorXd::ConstFixedSegmentReturnType<NV>::Type & q_dot = vs.segment<NV> (idx_v ());
393  typedef Eigen::Matrix<double, 2, 2> Matrix22;
394  typedef Eigen::Matrix<double, 2, 1> Vector2;
395 
396  const double& c0 = q(2), s0 = q(3);
397  Matrix22 R0;
398  R0 << c0, -s0, s0, c0;
399 
400  const double& t = q_dot[2];
401  const double theta = std::fabs(t);
402 
403  ConfigVector_t res;
404  if(theta > 1e-14)
405  {
406  // q_dot = [ x, y, t ]
407  // w = [ 0, 0, t ]
408  // v = [ x, y, 0 ]
409  // Considering only the 2x2 top left corner:
410  // Sp = [ 0, -1; 1, 0],
411  // if t > 0: S = Sp
412  // else : S = -Sp
413  // S / t = Sp / |t|
414  // S * S = - I2
415  // R = I2 + ( 1 - ct) / |t| * S + ( 1 - st / |t| ) * S * S
416  // = ( 1 - ct) / |t| * S + st / |t| * I2
417  //
418  // Ru = exp3 (w)
419  // tu = R * v = (1 - ct) / |t| * S * v + st / t * v
420  //
421  // M0 * Mu = ( R0 * Ru, R0 * tu + t0 )
422 
423  Eigen::VectorXd::ConstFixedSegmentReturnType<2>::Type v = vs.segment<2>(idx_v());
424  Eigen::Matrix<Scalar,2,1> cst;
425  SINCOS (t, &cst[1], &cst[0]);
426  const double inv_theta = 1/theta;
427  const double c_coeff = (1.-cst[0]) * inv_theta;
428  const double s_coeff = std::fabs(cst[1]) * inv_theta;
429  const Vector2 Sp_v (-v[1], v[0]);
430 
431  if (t > 0) res.head<2>() = q.head<2>() + R0 * (s_coeff * v + c_coeff * Sp_v);
432  else res.head<2>() = q.head<2>() + R0 * (s_coeff * v - c_coeff * Sp_v);
433  res.tail<2>() = R0 * cst;
434  return res;
435  }
436  else
437  {
438  // cos(t) ~ 1 - t^2 / 2
439  // sin(t) ~ t
440  res.head<2>() = q.head<2>() + R0*q_dot.head<2>();
441  res(2) = c0 * 1 - s0 * t;
442  res(3) = s0 * 1 + c0 * t;
443  }
444 
445  return res;
446  }
